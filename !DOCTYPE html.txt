let nextStaffId = 11;
let currentNoteKey = null;
const STORAGE_KEY = "dutyRosterState_v1";

function loadState(){
  try{
    const raw = localStorage.getItem(STORAGE_KEY);
    if(raw){
      const parsed = JSON.parse(raw);
      if(parsed && typeof parsed === "object"){
        state = Object.assign({}, state, parsed);
        if(parsed.staff && parsed.staff.length){
          nextStaffId = Math.max(...parsed.staff.map(s=>s.id)) + 1;
        }
      }
    }
  }catch(e){ console.warn("Could not load saved roster:", e); }
}

let saveTimer = null;
function saveState(){
  try{
    localStorage.setItem(STORAGE_KEY, JSON.stringify(state));
  }catch(e){ /* ignore - sync below is the source of truth when online */ }
  sendStateToServer();
  updateSaveStatus();
}
function markUnsaved(){
  if(applyingRemoteUpdate) return;
  const el = document.getElementById("saveStatus");
  if(el){ el.textContent = "Saving…"; el.style.color = "#8a8579"; }
  clearTimeout(saveTimer);
  saveTimer = setTimeout(saveState, 300);
}

// ---------------- Live sync (WebSocket via Cloudflare Durable Object) ----------------
const clientId = Math.random().toString(36).slice(2);
let ws = null;
let wsConnected = false;
let applyingRemoteUpdate = false; // true while we're applying a state update that came FROM the server, so we don't immediately echo it back
let reconnectDelay = 1000;

function connectSync(){
  const proto = location.protocol === "https:" ? "wss" : "ws";
  ws = new WebSocket(proto + "://" + location.host + "/ws");

  ws.onopen = () => {
    wsConnected = true;
    reconnectDelay = 1000;
    updateSaveStatus();
  };

  ws.onmessage = (ev) => {
    let msg;
    try{ msg = JSON.parse(ev.data); }catch(e){ return; }
    if((msg.type === "init" || msg.type === "update")){
      if(msg.state){
        applyingRemoteUpdate = true;
        state = msg.state;
        if(state.staff && state.staff.length){
          nextStaffId = Math.max(...state.staff.map(s=>s.id)) + 1;
        }
        recalcNoteIdCounter();
        renderAll();
        try{ localStorage.setItem(STORAGE_KEY, JSON.stringify(state)); }catch(e){}
        applyingRemoteUpdate = false;
      } else if(msg.type === "init" && !msg.state){
        // Brand new room, nothing on the server yet - push what we have locally.
        sendStateToServer();
      }
    }
    updateSaveStatus();
  };

  ws.onclose = () => {
    wsConnected = false;
    updateSaveStatus();
    setTimeout(connectSync, reconnectDelay);
    reconnectDelay = Math.min(reconnectDelay * 1.5, 15000);
  };

  ws.onerror = () => { wsConnected = false; updateSaveStatus(); };
}

function sendStateToServer(){
  if(applyingRemoteUpdate) return; // don't echo back what we just received
  if(ws && ws.readyState === WebSocket.OPEN){
    ws.send(JSON.stringify({ type: "update", state, clientId }));
  }
}

function updateSaveStatus(){
  const el = document.getElementById("saveStatus");
  if(!el) return;
  if(wsConnected){
    el.textContent = "● Live - synced " + new Date().toLocaleTimeString();
    el.style.color = "#2f6f63";
  } else {
    el.textContent = "○ Offline - saved to this device only";
    el.style.color = "#b85c5c";
  }
}

function cellKey(staffId, m, d){ return staffId+"-"+m+"-"+d; }
function daysInMonth(year, m){ return new Date(year, m+1, 0).getDate(); }
function isWeekend(year, m, d){ const wd = new Date(year, m, d).getDay(); return wd===0||wd===6; }
function holidayName(year, m, d){
  const iso = year+"-"+String(m+1).padStart(2,"0")+"-"+String(d).padStart(2,"0");
  const h = state.holidays.find(x=>x.date===iso);
  return h ? h.name : null;
}

function colorFor(val){
  if(!val) return null;
  const v = val.trim();
  if(!v) return null;
  const custom = state.shiftTypes.customCodes.find(c=>c.code.toLowerCase()===v.toLowerCase());
  if(custom) return custom.color;
  const lv = v.toLowerCase();
  const kc = state.shiftTypes.keywordColors;
  if(lv === "off wkend") return kc.offwkend;
  if(lv === "wkend") return kc.wkend;
  if(lv === "off") return kc.off;
  if(lv.includes("wfh")) return kc.wfh;
  if(lv.includes("al")) return kc.al;
  return null;
}

function workType(val){
  if(!val) return null;
  const lv = val.trim().toLowerCase();
  if(!lv) return null;
  if(lv.includes("wfh")) return "WFH";
  if(lv.includes("off") || lv.includes("al") || lv.includes("mc")) return null;
  return "WIO";
}

function allCodeSuggestions(){
  const base = ["AM","PM","Off","AL","AM AL","PM AL","WFH","AM WFH","PM WFH","Wkend","Off Wkend"];
  const custom = state.shiftTypes.customCodes.map(c=>c.code);
  return [...new Set([...base, ...custom])];
}

document.getElementById("applyBulk").addEventListener("click", ()=>{
  const ids = Array.from(document.querySelectorAll(".bulkStaffCheck:checked")).map(cb=>Number(cb.dataset.id));
  const from = Math.max(1, parseInt(document.getElementById("bulkFrom").value,10) || 1);
  const nd = daysInMonth(state.year, state.activeMonth);
  const to = Math.min(nd, parseInt(document.getElementById("bulkTo").value,10) || nd);
  const code = document.getElementById("bulkCode").value;
  const skipWeekend = document.getElementById("bulkSkipWeekend").checked;
  if(ids.length===0){ alert("Pick at least one staff member."); return; }
  if(!code){ alert("Enter a shift code to apply."); return; }
  ids.forEach(id=>{
    for(let d=from; d<=to; d++){
      if(skipWeekend && isWeekend(state.year, state.activeMonth, d)) continue;
      state.schedule[cellKey(id, state.activeMonth, d)] = code;
    }
  });
  renderTable();
  markUnsaved();
});

function computeStats(scope){
  const rows = state.staff.map(s=>({id:s.id, name:s.name, wfh:0, wio:0}));
  const months = scope==="year" ? [...Array(12).keys()] : [state.activeMonth];
  months.forEach(m=>{
    const nd = daysInMonth(state.year, m);
    rows.forEach(r=>{
      for(let d=1; d<=nd; d++){
        const val = state.schedule[cellKey(r.id, m, d)];
        const t = workType(val);
        if(t==="WFH") r.wfh++;
        else if(t==="WIO") r.wio++;
      }
    });
  });
  return rows;
}

function renderStats(){
  const scope = document.querySelector('input[name=statsScope]:checked').value;
  const rows = computeStats(scope);
  const totalWfh = rows.reduce((a,r)=>a+r.wfh,0);
  const totalWio = rows.reduce((a,r)=>a+r.wio,0);
  const total = totalWfh+totalWio;
  const wfhPct = total ? Math.round(totalWfh/total*100) : 0;
  const wioPct = total ? 100-wfhPct : 0;

  document.getElementById("statsSummary").innerHTML = `
    <div style="font-size:13px;margin-bottom:4px;"><b>${total}</b> worked entries · WFH <b>${totalWfh}</b> (${wfhPct}%) · WIO <b>${totalWio}</b> (${wioPct}%)</div>
    <div class="statsbar">
      <div style="width:${wfhPct}%;background:${state.shiftTypes.keywordColors.wfh}"></div>
      <div style="width:${wioPct}%;background:#cfe3d6"></div>
    </div>`;

  let html = "<tr><th>Staff</th><th>WFH days</th><th>WIO days</th><th>WFH %</th></tr>";
  rows.forEach(r=>{
    const t = r.wfh+r.wio;
    const pct = t ? Math.round(r.wfh/t*100) : 0;
    html += `<tr><td>${r.name}</td><td>${r.wfh}</td><td>${r.wio}</td><td>${t? pct+"%":"–"}</td></tr>`;
  });
  document.getElementById("statsTable").innerHTML = html;
}
document.querySelectorAll('input[name=statsScope]').forEach(r=>r.addEventListener("change", renderStats));

function exportExcel(){
  const wb = XLSX.utils.book_new();
  for(let m=0; m<12; m++){
    const nd = daysInMonth(state.year, m);
    const header = ["Name"];
    for(let d=1; d<=nd; d++){
      const wname = new Date(state.year,m,d).toLocaleDateString(undefined,{weekday:"short"});
      header.push(d+" "+wname);
    }
    const aoa = [header];
    state.staff.forEach(s=>{
      const row = [s.name];
      for(let d=1; d<=nd; d++){
        row.push(state.schedule[cellKey(s.id,m,d)] || "");
      }
      aoa.push(row);
    });
    aoa.push([]);
    const monthNoteList = getMonthNotes(m);
    if(monthNoteList.length){
      aoa.push(["Notes:"]);
      monthNoteList.forEach(n=> aoa.push([n.time || "", n.text]));
    }
    const ws = XLSX.utils.aoa_to_sheet(aoa);
    XLSX.utils.book_append_sheet(wb, ws, MONTHS[m]+" "+state.year);
  }

  // Notes sheet
  const noteRows = [["Date","Staff","Note"]];
  Object.keys(state.notes).forEach(key=>{
    const [sid,m,d] = key.split("-").map(Number);
    const staffObj = state.staff.find(s=>s.id===sid);
    if(!staffObj) return;
    const dateStr = state.year+"-"+String(m+1).padStart(2,"0")+"-"+String(d).padStart(2,"0");
    noteRows.push([dateStr, staffObj.name, state.notes[key]]);
  });
  if(noteRows.length>1){
    const wsNotes = XLSX.utils.aoa_to_sheet(noteRows);
    XLSX.utils.book_append_sheet(wb, wsNotes, "Notes");
  }

  // Stats sheet (whole year)
  const statsRows = computeStats("year");
  const statsAoa = [["Staff","WFH days","WIO days","WFH %"]];
  statsRows.forEach(r=>{
    const t = r.wfh+r.wio;
    statsAoa.push([r.name, r.wfh, r.wio, t? Math.round(r.wfh/t*100)+"%":"–"]);
  });
  const wsStats = XLSX.utils.aoa_to_sheet(statsAoa);
  XLSX.utils.book_append_sheet(wb, wsStats, "WFH-WIO Stats");

  XLSX.writeFile(wb, "Duty_Roster_"+state.year+".xlsx");
}

let noteIdCounter = 1;
function getMonthNotes(m){
  const entry = state.monthNotes[m];
  if(!entry) return [];
  if(typeof entry === "string"){
    // migrate legacy single-string note to the new list format
    const migrated = entry.trim() ? [{id:noteIdCounter++, text:entry, time:""}] : [];
    state.monthNotes[m] = migrated;
    return migrated;
  }
  return entry;
}

function renderMonthNotes(){
  document.getElementById("monthNotesTitle").textContent = "Notes for " + MONTHS_FULL[state.activeMonth] + " " + state.year;
  const notes = getMonthNotes(state.activeMonth);
  const wrap = document.getElementById("monthNotesList");
  if(notes.length === 0){
    wrap.innerHTML = `<div class="noteempty">No notes yet for this month.</div>`;
    return;
  }
  wrap.innerHTML = `<div class="notelist">` + notes.map(n => `
    <div class="noteitem">
      <div class="noteitemtext">${n.text.replace(/</g,"&lt;")}${n.time?`<span class="notetime">${n.time}</span>`:""}</div>
      <button data-id="${n.id}" title="Delete note">×</button>
    </div>`).join("") + `</div>`;
  wrap.querySelectorAll("button[data-id]").forEach(btn=>{
    btn.addEventListener("click", e=>{
      const id = Number(e.target.dataset.id);
      state.monthNotes[state.activeMonth] = getMonthNotes(state.activeMonth).filter(n=>n.id!==id);
      renderMonthNotes();
      markUnsaved();
    });
  });
}

function addMonthNote(){
  const input = document.getElementById("newNoteText");
  const text = input.value.trim();
  if(!text) return;
  const list = getMonthNotes(state.activeMonth);
  list.push({
    id: noteIdCounter++,
    text,
    time: new Date().toLocaleString([], {month:"short", day:"numeric", hour:"2-digit", minute:"2-digit"})
  });
  state.monthNotes[state.activeMonth] = list;
  input.value = "";
  renderMonthNotes();
  markUnsaved();
}

document.getElementById("restoreBackupFile").addEventListener("change", e=>{
  const file = e.target.files[0];
  if(!file) return;
  const reader = new FileReader();
  reader.onload = ev=>{
    try{
      const parsed = JSON.parse(ev.target.result);
      state = Object.assign({}, state, parsed);
      if(parsed.staff && parsed.staff.length){
        nextStaffId = Math.max(...parsed.staff.map(s=>s.id)) + 1;
      }
      recalcNoteIdCounter();
      renderAll();
      alert("Backup restored.");
    }catch(err){
      alert("Could not read that file - is it a Duty Roster backup .json?");
    }
  };
  reader.readAsText(file);
  e.target.value = "";
});

function recalcNoteIdCounter(){
  let maxId = 0;
  Object.values(state.monthNotes || {}).forEach(entry=>{
    if(Array.isArray(entry)) entry.forEach(n=>{ if(n.id>maxId) maxId = n.id; });
  });
  noteIdCounter = maxId + 1;
}
